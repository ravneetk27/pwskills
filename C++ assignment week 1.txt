ans 4 . WAP for finding the volume of the cylinder by taking radius and height as input.

#include<iostream>
using namespace std;
int main(){
    int r;
    int h;
    cout<<"enter radius of cylinder : ";
    cin>>r;
    cout<<"enter height of cylinder : ";
    cin>>h;
    int volume = 3.14*r*r*h;
    cout<<"volume of cylinder : "<<volume<<endl;
}
